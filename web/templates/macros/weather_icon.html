{# weather_icon macro — phong cách weather.com Params: weather_main : str — OpenWeatherMap weather_main field (e.g. "Clear", "Rain", ...) clouds : int — Cloud cover % (0-100), default 50 ts_utc : datetime | None — UTC timestamp để xác định ngày/đêm (ICT = UTC+7) size : str — Tailwind class kích thước, e.g. "w-10 h-10" #} {% macro icon(weather_main, clouds=50, ts_utc=None, size="w-10 h-10") %} {#- Xác định ngày/đêm từ giờ ICT (UTC+7) -#} {% if ts_utc and ts_utc.hour is defined %} {% set hour_ict = (ts_utc.hour + 7) % 24 %} {% else %} {% set hour_ict = 12 %} {% endif %} {% set is_day = hour_ict >= 6 and hour_ict < 18 %} {#- Chọn icon name dựa trên weather_main -#} {% set wm = weather_main | default("") %} {% if wm == "Clear" %} {% set icon_name = "clear-day" if is_day else "clear-night" %} {% elif wm == "Clouds" %} {% if (clouds or 50) | int < 50 %} {% set icon_name = "partly-cloudy-day" if is_day else "partly-cloudy-night" %} {% else %} {% set icon_name = "cloudy" %} {% endif %} {% elif wm == "Rain" %} {% set icon_name = "rain" %} {% elif wm == "Drizzle" %} {% set icon_name = "drizzle" %} {% elif wm == "Thunderstorm" %} {% set icon_name = "thunderstorm" %} {% elif wm == "Snow" %} {% set icon_name = "snow" %} {% elif wm in ["Mist", "Fog", "Haze", "Smoke", "Dust", "Sand", "Ash", "Squall"] %} {% set icon_name = "fog" %} {% elif wm in ["Wind", "Tornado"] %} {% set icon_name = "wind" %} {% else %} {% set icon_name = "partly-cloudy-day" if is_day else "partly-cloudy-night" %} {% endif %} {% endmacro %}
